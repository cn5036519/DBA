# 13. 输入一个字符串，要求判断在这个字符串中大写字母，小写字母，数字，其它字符共出现了多少次，并输出来
number_count = uppercase_count = lowercase_count = special_characters_count = 0
data = input("请输入一个字符串:")
for x in data:
    if x.isdigit():
        number_count += 1
    elif x.isalpha():
        if x.isupper():
            uppercase_count += 1
        else:
            lowercase_count += 1
    else:
        special_characters_count += 1
print(number_count)
print(lowercase_count)
print(uppercase_count)
print(special_characters_count)
